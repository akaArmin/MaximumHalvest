### Option 1: Python-Script (einfach & lokal)

```python
import yfinance as yf
import pandas as pd
from datetime import datetime

# ── Konfiguration ───────────────────────────────────────────
SELL_TICKER   = "O"          # Realty Income
BUY_TICKER    = "XXXX"       # ETF-Ticker eintragen
SHARES        = 60
ALERT_SELL_ABOVE = 55.00     # Kurs-Ziel Verkauf (USD)
ALERT_BUY_BELOW  = 100.00    # Kurs-Ziel Kauf   (EUR/USD)

# ── Daten abrufen ───────────────────────────────────────────
def get_price(ticker: str) -> dict:
    stock = yf.Ticker(ticker)
    info  = stock.fast_info
    return {
        "ticker"  : ticker,
        "price"   : round(info.last_price, 2),
        "currency": info.currency,
        "time"    : datetime.now().strftime("%Y-%m-%d %H:%M"),
    }

# ── Bewertung ───────────────────────────────────────────────
def evaluate(sell: dict, buy: dict):
    total_sell = sell["price"] * SHARES
    
    print("\n" + "="*50)
    print(f"📅  {sell['time']}")
    print(f"\n📈  VERKAUF  | {sell['ticker']}")
    print(f"    Kurs     : {sell['price']} {sell['currency']}")
    print(f"    60 Stück : {total_sell:.2f} {sell['currency']}")
    print(f"    Signal   : {'✅ ÜBER Ziel' if sell['price'] >= ALERT_SELL_ABOVE else '⏳ noch abwarten'}")
    
    print(f"\n📉  KAUF     | {buy['ticker']}")
    print(f"    Kurs     : {buy['price']} {buy['currency']}")
    print(f"    Signal   : {'✅ UNTER Ziel' if buy['price'] <= ALERT_BUY_BELOW else '⏳ noch abwarten'}")
    
    if sell["price"] >= ALERT_SELL_ABOVE and buy["price"] <= ALERT_BUY_BELOW:
        print("\n🚀  >>> OPTIMALER ZEITPUNKT ERREICHT! <<<")
    print("="*50)

# ── Hauptprogramm ───────────────────────────────────────────
if __name__ == "__main__":
    sell_data = get_price(SELL_TICKER)
    buy_data  = get_price(BUY_TICKER)
    evaluate(sell_data, buy_data)
```

## Kern-Logik

```python
# tracker.py
import yfinance as yf
import requests
from datetime import datetime
from config import CONFIG

def get_prices() -> dict:
    """Kurse & Wechselkurs abrufen"""
    
    sell  = yf.Ticker(CONFIG["SELL_TICKER"])
    buy   = yf.Ticker(CONFIG["BUY_TICKER"])
    fx    = yf.Ticker("EURUSD=X")

    sell_price = sell.fast_info.last_price
    buy_price  = buy.fast_info.last_price
    eurusd     = fx.fast_info.last_price      # z.B. 1.08

    # O notiert in USD → umrechnen in EUR
    sell_price_eur = sell_price / eurusd
    total_eur      = sell_price_eur * CONFIG["SHARES_SELL"]
    etf_shares     = total_eur / buy_price

    return {
        "timestamp"      : datetime.now().strftime("%Y-%m-%d %H:%M"),
        "O_usd"          : round(sell_price, 4),
        "O_eur"          : round(sell_price_eur, 4),
        "total_eur"      : round(total_eur, 2),
        "XDWD_eur"       : round(buy_price, 4),
        "etf_shares"     : round(etf_shares, 4),
        "etf_shares_full": int(etf_shares),   # kaufbare ganze Anteile
        "eurusd"         : round(eurusd, 4),
    }


def send_signal_message(text: str):
    """Nachricht via signal-cli REST API senden"""
    if not CONFIG["SIGNAL_ENABLE"]:
        return
    
    payload = {
        "message"   : text,
        "number"    : CONFIG["SIGNAL_SENDER"],
        "recipients": [CONFIG["SIGNAL_RECEIVER"]],
    }
    try:
        requests.post(
            f"{CONFIG['SIGNAL_URL']}/v2/send",
            json=payload,
            timeout=5
        )
    except Exception as e:
        print(f"Signal-Fehler: {e}")


def format_message(data: dict, prev_best: int) -> str:
    """Nachrichtentext zusammenstellen"""
    
    trend = "🆕 NEUES MAXIMUM" if data["etf_shares_full"] > prev_best else ""
    
    return (
        f"📊 Realty Income → MSCI World Tracker\n"
        f"{'─'*35}\n"
        f"🕐 {data['timestamp']}\n\n"
        f"📈 Realty Income (O)\n"
        f"   {data['O_usd']} USD  ({data['O_eur']} EUR)\n"
        f"   60 Stück = {data['total_eur']} EUR\n\n"
        f"📉 Xtrackers MSCI World\n"
        f"   {data['XDWD_eur']} EUR\n\n"
        f"💱 EUR/USD: {data['eurusd']}\n\n"
        f"✅ Kaufbare ETF-Anteile: {data['etf_shares_full']}\n"
        f"   (exakt: {data['etf_shares']})\n"
        f"{trend}"
    )
```

## Scheduler mit Benachrichtigung bei neuem Maximum

```python
# main.py
import schedule
import time
import json
import os
from tracker import get_prices, send_signal_message, format_message
from config import CONFIG

HISTORY_FILE = "history.json"

def load_history() -> list:
    if os.path.exists(HISTORY_FILE):
        with open(HISTORY_FILE) as f:
            return json.load(f)
    return []

def save_history(history: list):
    with open(HISTORY_FILE, "w") as f:
        json.dump(history, f, indent=2)

def job():
    history = load_history()
    prev_best = max((h["etf_shares_full"] for h in history), default=0)
    
    data = get_prices()
    history.append(data)
    save_history(history)
    
    msg = format_message(data, prev_best)
    print(msg)
    
    # Signal senden wenn:
    # a) Neues Maximum ODER
    # b) Morgens um 9 Uhr als tägliches Update
    is_new_max   = data["etf_shares_full"] > prev_best
    is_daily_sum = datetime.now().hour == 9 and datetime.now().minute < 31
    
    if is_new_max or is_daily_sum:
        send_signal_message(msg)

# ── Start ────────────────────────────────────────────────────
schedule.every(CONFIG["CHECK_INTERVAL_MIN"]).minutes.do(job)
schedule.every().day.at("09:00").do(job)   # tägliche Zusammenfassung

print("🚀 Tracker gestartet...")
job()  # direkt einmal ausführen

while True:
    schedule.run_pending()
    time.sleep(1)
```

### Angepasstes Tracker Script für GitHub Actions

```python
# tracker.py – optimiert für GitHub Actions
import yfinance as yf
import requests
import json
import os
from datetime import datetime

# ── Konfiguration ────────────────────────────────────────────
SHARES_SELL = 60
SELL_TICKER = "O"
BUY_TICKER  = "XDWD.DE"

# Telegram (einfacher als Signal!)
TG_TOKEN   = os.environ.get("TELEGRAM_TOKEN")
TG_CHAT_ID = os.environ.get("TELEGRAM_CHAT_ID")

# ── Kurse abrufen ────────────────────────────────────────────
def get_prices() -> dict:
    sell  = yf.Ticker(SELL_TICKER)
    buy   = yf.Ticker(BUY_TICKER)
    fx    = yf.Ticker("EURUSD=X")

    sell_price = sell.fast_info.last_price
    buy_price  = buy.fast_info.last_price
    eurusd     = fx.fast_info.last_price

    sell_eur   = sell_price / eurusd
    total_eur  = sell_eur * SHARES_SELL
    etf_shares = total_eur / buy_price

    return {
        "timestamp"      : datetime.now().strftime("%Y-%m-%d %H:%M"),
        "O_usd"          : round(sell_price, 2),
        "O_eur"          : round(sell_eur, 2),
        "total_eur"      : round(total_eur, 2),
        "XDWD_eur"       : round(buy_price, 2),
        "etf_shares"     : round(etf_shares, 4),
        "etf_shares_full": int(etf_shares),
        "eurusd"         : round(eurusd, 4),
    }

# ── Bisheriges Maximum laden ─────────────────────────────────
def load_best() -> int:
    """
    Wir speichern das Maximum als GitHub Actions Variable
    Alternativ: in einer kleinen Datei im Repo
    """
    try:
        with open("best.json") as f:
            return json.load(f)["best"]
    except:
        return 0

def save_best(value: int):
    with open("best.json", "w") as f:
        json.dump({"best": value, 
                   "updated": datetime.now().strftime("%Y-%m-%d %H:%M")}, f)

# ── Telegram Nachricht ───────────────────────────────────────
def send_telegram(text: str):
    if not TG_TOKEN:
        print("Kein Telegram Token – nur Konsole")
        return
    
    url = f"https://api.telegram.org/bot{TG_TOKEN}/sendMessage"
    requests.post(url, json={
        "chat_id"   : TG_CHAT_ID,
        "text"      : text,
        "parse_mode": "Markdown"
    })

def format_message(data: dict, is_new_max: bool) -> str:
    header = "🏆 *NEUES MAXIMUM!*\n\n" if is_new_max else "📊 *Tages-Update*\n\n"
    
    return (
        f"{header}"
        f"🕐 {data['timestamp']}\n\n"
        f"📈 *Realty Income (O)*\n"
        f"   `{data['O_usd']} USD` → `{data['O_eur']} EUR`\n"
        f"   60 Stück = `{data['total_eur']} EUR`\n\n"
        f"📉 *Xtrackers MSCI World*\n"
        f"   `{data['XDWD_eur']} EUR`\n\n"
        f"💱 EUR/USD: `{data['eurusd']}`\n\n"
        f"✅ *Kaufbare ETF-Anteile: {data['etf_shares_full']}*\n"
        f"   _(exakt: {data['etf_shares']})_"
    )

# ── Hauptprogramm ────────────────────────────────────────────
if __name__ == "__main__":
    data     = get_prices()
    prev_best = load_best()
    is_new_max = data["etf_shares_full"] > prev_best
    
    print(format_message(data, is_new_max))
    
    # Immer um 9 Uhr senden (erster Run des Tages)
    current_hour = datetime.now().hour
    is_morning   = current_hour == 8  # UTC = 9 Uhr MEZ
    
    if is_new_max or is_morning:
        send_telegram(format_message(data, is_new_max))
    
    # Neues Maximum speichern
    if is_new_max:
        save_best(data["etf_shares_full"])
        # Datei ins Repo committen
        os.system('git config user.email "tracker@bot.com"')
        os.system('git config user.name "Tracker Bot"')
        os.system('git add best.json')
        os.system(f'git commit -m "New best: {data["etf_shares_full"]} ETF shares"')
        os.system('git push')
```

         ↓
